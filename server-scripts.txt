SCRIPT FOR DOCKER INSTALLATION:

#!/bin/bash

# Install Docker
## Update index of `apt` package
sudo apt-get update

## Install packages, to allow `apt` use repositoty using HTTPS
sudo apt-get install -y apt-transport-https ca-certificates curl gnupg-agent software-properties-common

## Add official GPG-key Docker
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo apt-key add -

## Add official GPG-key Docker
sudo add-apt-repository "deb [arch=amd64] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable"

## Update index of `apt` package (again)
sudo apt-get update

## Install latest version of Docker
sudo apt-get install -y docker-ce docker-ce-cli containerd.io

# Install Docker Compose
## Download latest version of Docker Compose
sudo apt-get install docker-compose-plugin

## Make Docker Compose executable
sudo chmod +x /usr/local/bin/docker-compose

## Add symbolic link to executable file Docker Compose in /usr/bin
sudo ln -s /usr/local/bin/docker-compose /usr/bin/docker-compose

## Check
sudo docker --version
sudo docker compose version

## Add user into group Docker
sudo usermod -aG docker $USER

## Reboot
sudo reboot
